# backend app server start
uvicorn backend.views:app --reload

# backend update dependencies
pip-compile --generate-hashes backend/requirements/production.in
pip-compile --generate-hashes backend/requirements/production.in backend/requirements/develop.in --output-file backend/requirements/develop.txt
pip-sync backend/requirements/production.txt
pip-sync backend/requirements/develop.txt

# frontend
cd frontend
npm install
npm run dev
